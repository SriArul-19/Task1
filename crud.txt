AdminController

@PostMapping("/findCustomer")
    public String findCustomerById(@RequestParam("customerId") Long customerId, Model model) {
        Customer customer = customerService.getCustomerById(customerId);

        if (customer == null) {
            model.addAttribute("message", "Customer ID not found!");
            return "CustomerResult"; // result page
        }

        model.addAttribute("customer", customer);
        return "CustomerResult";  // show results
    }

@GetMapping("/customers")
public String getAllCustomers(Model model) {
    List<Customer> customers = customerService.getAllCustomer();
    model.addAttribute("customers", customers);
    return "GetAllCustomer";   // JSP page name (GetAllCustomer.jsp)
}
